#include <MaxMatrix.h>

//изображение буквы А размером - в высоту 8 и в ширину 4
byte CH_A[] = {4, 8, B11111110, B00010001, B00010001, B11111110 }; 

MaxMatrix m(8, 9, 10, 1);

void setup(){
  m.init();
  m.setIntensity(0); //Яркость 0-15, где 15 максимум, а 0 тусклое свечение
}

void loop(){
 m.writeSprite(0, 0, CH_A);
}
